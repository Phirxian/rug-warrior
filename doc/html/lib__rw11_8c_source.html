188 void motor(int index, float vel)
189 { float avel ; /* Absolute value of velocity */
190  if (vel > 0.0)
191  { bit_set(0x1008, dir_mask[index]); /* Forward rotation */
192  avel = vel; }
193  else
194  { bit_clear(0x1008, dir_mask[index]); /* Backward rotation */
195  avel = (- vel); }
196  if (avel < 1.0) /* If we are going real slow */
197  bit_clear(0x100D,pwm_mask[index]); /* then just have OC1 turn off the motor */
198  else
199  bit_set(0x100D,pwm_mask[index]); /* Otherwise let OC1 control the motors */
200  if (avel > 99.0) /* If we are gong real fast */
201  avel = 99.0; /* don't let the following multiply overflow */
202  pokeword(TOCx[index], (int) (655.56 * avel)); /* Compute TOCx value */
203 }

220 int bumper()
221 { int bmpr;
222  bmpr = analog(3); /* Switch closure: */
223  if (bmpr < 11) return 0b000; /* none */
224  else if (bmpr < 32) return 0b001; /* A */

237 int ir_detect()
238 { int val1, val2, val3;
239  val1 = peek(0x100A) & 0b10000; /* IR Detector connected to PE4 */
240  bit_set(0x1008,0b1000); /* Turn on Left emitter, PD3 */
241  msleep(1L); /* Wait 1 millisecond */
242  val2 = peek(0x100A) & 0b10000; /* Should be Low if signal detected */
243  bit_clear(0x1008,0b1000); /* Turn off Left emitter */
244  bit_set(0x1008,0b0100); /* Turn on Right emitter, PD2 */
245  msleep(1L); /* Wait 1 millisecond */
246  val3 = peek(0x100A) & 0b10000;
247  bit_clear(0x1008,0b1100); /* Turn emitters off */
248  /* For detection, detector must be high when emitter is off, low when on */
249  return ((val1 & ~val2) >> 3) | ((val1 & ~val3) >> 4); /* HI -> LOW */
250 }
