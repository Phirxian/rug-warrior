153 {

155  bmpr = analog(3); /* Switch closure: */
156 
157  if(bmpr < 11) return 0b000; /* none */
158  else if(bmpr < 32) return 0b001; /* A */
159  else if(bmpr < 53) return 0b010; /* B */
160  else if(bmpr < 74) return 0b011; /* AB */
161  else if(bmpr < 96) return 0b100; /* C */
162  else if(bmpr < 117) return 0b101; /* A C */

164  else return 0b111; /* ABC - (mechanically impossible) */

177 int ir_detect()
178 
179 {
180  int val1, val2, val3;
181  val1 = peek(0x100A) & 0b10000; /* IR Detector connected to PE4 */
182  bit_set(0x1008,0b0100); /* Turn on Left emitter, PD2 */
183  msleep(1L); /* Wait 1 millisecond */
184  val2 = peek(0x100A) & 0b10000; /* Should be Low if signal detected */
185  bit_clear(0x1008,0b0100); /* Turn off Left emitter */
186  bit_set(0x1008,0b1000); /* Turn on Right emitter, PD3 */
187  msleep(1L); /* Wait 1 millisecond */
188  val3 = peek(0x100A) & 0b10000;
189  bit_clear(0x1008,0b1100); /* Turn emitters off */
190  /* For detection, detector must be high when emitter is off, low when on */
191  return ((val1 & ~val2) >> 3) | ((val1 & ~val3) >> 4); /* HI -> LOW */
192 }
