401 int digital(int port)
402 {
403  if (port==0) /* this is first bit of PORTA */
404  return !(peek(0x1000)&1);
405  if (port==1) /* this is the second bit of PORTA */
406  return !(peek(0x1000)&2);
407  if (port<8) /* normal bit of 7fff as gotten from digital chip */
408  return !((peek(0x7fff)>>port)&1);
409  if(port<36)
