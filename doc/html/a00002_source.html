139 int ir_detect()

143  bit_set(0x1008,0b0100); /* Turn on Left emitter, PD2 */
144  msleep(1L); /* Wait 1 millisecond */

146  bit_clear(0x1008,0b0100); /* Turn off Left emitter */
147  bit_set(0x1008,0b1000); /* Turn on Right emitter, PD3 */
148  msleep(1L); /* Wait 1 millisecond */
