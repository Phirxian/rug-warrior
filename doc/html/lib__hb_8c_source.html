245 void _set_motor(int motor, int dir, int speed)
246 {
247  bit_set(0x0e, 1 << (4 + motor)); /* turn motor on */
248  if (dir)
249  bit_set(0x0e, 1 << motor); /* set direction for backward */
250  else
251  bit_clear(0x0e, 1 << motor); /* set dir for forward */
252  _motor_speed[motor]= speed;
253  _set_motor_speeds();
254 }

256 void _set_motor_speeds()
257 {
258  int speed01= (_speed_table[_motor_speed[0]] << 8)
259  + _speed_table[_motor_speed[1]];
260  int speed23= (_speed_table[_motor_speed[2]] << 8)
261  + _speed_table[_motor_speed[3]];
262 
263  pokeword(0x22, speed01);
264  pokeword(0x24, speed23);
265 }

320 int digital(int port)
321 {
322  if (port < 7) /* analogs */
323  return analog(port) < 255;
324  if (port == 7) /* TIC1 */
325  return !(peek(0x1000) & 1);
326  if (port == 8) /* TIC2 */
327  return !(peek(0x1000) & 2);
328  if (port == 9) /* PAI */
329  return !(peek(0x1000) & 128);
330  if (port < 16) /* normal bit of 7fff as gotten from digital chip */
331  return !((peek(0x7fff) >> (port - 10)) & 1);
332  else {
333  printf("Digital port outof range\n");
334  beep();
335  return -1;
336  }
337 }
